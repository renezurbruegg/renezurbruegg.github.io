import { Injectable } from '@angular/core';
import {timer, from, Observable, Subject } from 'rxjs';
import { HttpClient } from '@angular/common/http';
import { switchMap } from 'rxjs/operators'
import { AppConfig } from '../_services/app.config';
import { ToastrService } from 'ngx-toastr';

@Injectable({ providedIn: 'root' })
export class VoltageUpdateService {

    private subjects:any = {};
    private pollUrl = AppConfig.apiUrl + "/voltageUpdate/ALL";
    private subscriberCounter = 0;
    private obs: Observable<any>;
    private subscription: any;
    private running: boolean = false;

    constructor(private http: HttpClient,  private toastr: ToastrService) {
    console.log("Volt update service grated")
   }

    _getSubjectObeservable(chartId:string) {
      console.log("get subj for " + chartId)
      //console.log(this.subjects)
      if(!this.subjects.hasOwnProperty(chartId)) {
        this.subjects[chartId] = new Subject<any>();
      }
      //console.log("subj:")
      //console.log(this.subjects)
      return this.subjects[chartId];
    }

    notifyUnsubscribe() {
        this.subscriberCounter-= 1;
        if(this.subscriberCounter == 0 && this.running)
          this.stop()
    }

    notifySubscribe() {
      this.subscriberCounter+= 1;
      if(this.subscriberCounter > 0 && !this.running)
        this.start()
    }

    sendVoltageUpdate(chartId:string, voltageValue:number) {
      //console.log("sending volt updt to  " + chartId + " value:" + voltageValue)
      this._getSubjectObeservable(chartId).next(voltageValue)
    }

    stop() {
      this.subscription.unsubscribe();
      this.running = false;
    }


    start() {
      console.log("Starting ")
      console.log("interval " + AppConfig.settings.general.timerInterval)
      this.obs  = timer(0, AppConfig.settings.general.timerInterval)
                    .pipe(
                      switchMap( () => from(this.http.get(this.pollUrl))),
                    )
      this.running = true;

      this.subscription = this.obs.subscribe((data) => {
        this.handlePolledData(data)
      },
      response => {
          console.log("Error while retreaving data", response);
          this.toastr.error(response, "Error while retreaving data");

      });

      //this.subscription = this.obs.subscribe((data) => this.handlePolledData(data))
    }

    handlePolledData(data: any) {
      for (let chartId in this.subjects) {
        this.subjects[chartId].next(data);
      }

      // for (let key in this.pollDataTransformer) {
      //   this._getSubjectObeservable(key).next(
      //     this.pollDataTransformer[key](data)
      //   );
      // }
    }

    clearMessage(chartId:string) {
        this._getSubjectObeservable(chartId).next();
    }

    getVoltageUpdateObs(chartId:string): Observable<any> {
        //console.log("gettin voltage obs")
          return this._getSubjectObeservable(chartId).asObservable();
    }

    // getVoltageUpdateWithCustomHandler(chartId, handlerFunction): Observable<any> {

    //       return this._getSubjectObeservable(chartId).asObservable();
    // }


}
